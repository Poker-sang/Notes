\paragraph{命题中}

\begin{enumerate}
    \item 以分号$;$分隔命题、作合取用
    \item 以逗号$,$分隔变量
    \item 定义中竖线$|$左边指定义的对象（一定是自由变元），右边是定义需要满足的命题
    \item 条件$\to$左边是前提，右边是需要满足的条件（一般优先级最低）
    \item 量词$\forall,\exists$指定了约束变元，最好不要省略；
          可以写为形如$\forall x\in A$或$\exists x\p{0>x>1}$两种形式
\end{enumerate}

$\ast,\triangle:A^2\to A;a,b\in A;\forall x,y,z\in A$

\begin{center}
    \begin{longtblr}{colspec={c|c|c},hlines}
        \hline
        名词             & 符号                                                                                                               & 定义或备注                                      \\

        封闭             & \SetCell[c=2]{c} $\ast|x\ast y\in A$                                                                               &                                                 \\
        交换             & \SetCell[c=2]{c} $\ast|x\ast y=y\ast x$                                                                            &                                                 \\
